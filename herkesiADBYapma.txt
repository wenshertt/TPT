local radius=5
local scanInterval=0.1
local teamName="As"
local highlightDefaultColor=Color3.fromRGB(0,255,0)
local Players=game:GetService("Players")
local RunService=game:GetService("RunService")
local UserInputService=game:GetService("UserInputService")
local TextChatService=game:GetService("TextChatService")
local LocalPlayer=Players.LocalPlayer
local Character=LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
local HRP=Character:WaitForChild("HumanoidRootPart")
local channel=TextChatService:FindFirstChild("TextChannels") and TextChatService.TextChannels:FindFirstChild("RBXGeneral")
local updateConnection
local scanningTask
local sendingMessages=false
local gui=Instance.new("ScreenGui",game.CoreGui)
gui.Name="HitboxUI"
local main=Instance.new("Frame",gui)
main.Size=UDim2.new(0,260,0,306)
main.Position=UDim2.new(1,-270,0.5,-153)
main.BackgroundColor3=Color3.fromRGB(15,15,15)
main.BackgroundTransparency=0.15
main.BorderSizePixel=0
main.Active=true
Instance.new("UICorner",main).CornerRadius=UDim.new(0,14)
local title=Instance.new("TextLabel",main)
title.Size=UDim2.new(1,0,0,36)
title.BackgroundColor3=Color3.fromRGB(25,25,25)
title.Text="Herkesi Asayiş D.B. Yapma"
title.Font=Enum.Font.GothamBold
title.TextSize=15
title.TextColor3=Color3.new(1,1,1)
title.TextXAlignment=Enum.TextXAlignment.Center
Instance.new("UICorner",title).CornerRadius=UDim.new(0,14)
local sydearrLabel=Instance.new("TextLabel",title)
sydearrLabel.Size=UDim2.new(0,70,0,15)
sydearrLabel.Position=UDim2.new(1,-75,1,-15)
sydearrLabel.BackgroundTransparency=1
sydearrLabel.Text="Sydearr Team"
sydearrLabel.Font=Enum.Font.GothamBold
sydearrLabel.TextSize=9
sydearrLabel.TextColor3=Color3.new(0.7,0.7,0.7)
sydearrLabel.TextXAlignment=Enum.TextXAlignment.Right
sydearrLabel.TextYAlignment=Enum.TextYAlignment.Bottom
local scroll=Instance.new("ScrollingFrame",main)
scroll.Size=UDim2.new(1,-20,1,-110)
scroll.Position=UDim2.new(0,10,0,40)
scroll.BackgroundTransparency=1
scroll.CanvasSize=UDim2.new(0,0,0,0)
scroll.ScrollBarThickness=5
local layout=Instance.new("UIListLayout",scroll)
layout.Padding=UDim.new(0,5)
layout.SortOrder=Enum.SortOrder.LayoutOrder
layout.HorizontalAlignment=Enum.HorizontalAlignment.Center
local bubble=Instance.new("Part",workspace)
bubble.Shape=Enum.PartType.Ball
bubble.Anchored=true
bubble.CanCollide=false
bubble.Transparency=0.85
bubble.Material=Enum.Material.ForceField
bubble.Color=Color3.fromRGB(0,255,255)
bubble.Size=Vector3.new(radius*2,radius*2,radius*2)
bubble.Name="PlayerBubble"
bubble.CastShadow=false
local isLocked=false
local function addHighlight(player)
if not player.Character or not player.Character:FindFirstChild("HumanoidRootPart") then return end
if player.Character:FindFirstChild("Highlight") then return end
local h=Instance.new("Highlight",player.Character)
h.FillColor=highlightDefaultColor
h.OutlineColor=Color3.fromRGB(255,255,255)
h.DepthMode=Enum.HighlightDepthMode.AlwaysOnTop
if not player.Character:FindFirstChild("NameTag") then
local tag=Instance.new("BillboardGui",player.Character)
tag.Name="NameTag"
tag.Adornee=player.Character:FindFirstChild("Head") or player.Character:FindFirstChild("HumanoidRootPart")
tag.Size=UDim2.new(0,100,0,25)
tag.StudsOffset=Vector3.new(0,2.5,0)
tag.AlwaysOnTop=true
local text=Instance.new("TextLabel",tag)
text.Size=UDim2.new(1,0,1,0)
text.BackgroundTransparency=1
text.Font=Enum.Font.GothamBold
text.TextSize=14
text.TextColor3=Color3.new(1,1,1)
text.Text=player.Name
end
end
local function clearHighlights()
for _,plr in ipairs(Players:GetPlayers())do
if plr.Character then
if plr.Character:FindFirstChild("Highlight")then
plr.Character.Highlight:Destroy()
end
if plr.Character:FindFirstChild("NameTag")then
plr.Character.NameTag:Destroy()
end
end
end
end
local function updateList()
if isLocked then return end
for _,b in ipairs(scroll:GetChildren())do
if b:IsA("TextButton")then b:Destroy()end
end
clearHighlights()
local count=0
for _,plr in ipairs(Players:GetPlayers())do
if plr~=LocalPlayer and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart")then
local dist=(HRP.Position-plr.Character.HumanoidRootPart.Position).Magnitude
if dist<=radius then
addHighlight(plr)
local btn=Instance.new("TextButton",scroll)
btn.Size=UDim2.new(1,-6,0,30)
btn.BackgroundColor3=Color3.fromRGB(0,255,0)
btn.Text=plr.Name
btn.TextColor3=Color3.new(1,1,1)
btn.Font=Enum.Font.GothamBold
btn.TextSize=14
btn.AutoButtonColor=false
Instance.new("UICorner",btn).CornerRadius=UDim.new(0,8)
count+=1
end
end
end
scroll.CanvasSize=UDim2.new(0,0,0,count*36)
end
local bottom=Instance.new("Frame",main)
bottom.Size=UDim2.new(1,-20,0,60)
bottom.Position=UDim2.new(0,10,1,-65)
bottom.BackgroundTransparency=1
local grid=Instance.new("UIGridLayout",bottom)
grid.CellSize=UDim2.new(0.32,0,0,26)
grid.CellPadding=UDim2.new(0.02,0,0,6)
grid.FillDirectionMaxCells=3
local function createBtn(text,color,cb)
local b=Instance.new("TextButton",bottom)
b.Text=text
b.BackgroundColor3=color
b.TextColor3=Color3.new(1,1,1)
b.Font=Enum.Font.GothamBold
b.TextSize=13
Instance.new("UICorner",b).CornerRadius=UDim.new(0,10)
b.MouseButton1Click:Connect(cb)
return b
end
createBtn("Büyüt",Color3.fromRGB(0,110,220),function()
radius+=2
bubble.Size=Vector3.new(radius*2,radius*2,radius*2)
end)
createBtn("Küçült",Color3.fromRGB(220,110,0),function()
radius=math.max(2,radius-2)
bubble.Size=Vector3.new(radius*2,radius*2,radius*2)
end)
local printBtn=Instance.new("TextButton",bottom)
printBtn.Text="Yazdır"
printBtn.BackgroundColor3=Color3.fromRGB(0,170,0)
printBtn.TextColor3=Color3.new(1,1,1)
printBtn.Font=Enum.Font.GothamBold
printBtn.TextSize=13
Instance.new("UICorner",printBtn).CornerRadius=UDim.new(0,10)
printBtn.Size=UDim2.new(0.32,0,0,26)
printBtn.Position=UDim2.new(0.34,0,0,33)
printBtn.MouseButton1Click:Connect(function()
if not channel then warn("Chat kanalı bulunamadı.") return end
sendingMessages=true
local children={}
for _,b in ipairs(scroll:GetChildren())do
if b:IsA("TextButton")then
table.insert(children,b)
end
end
task.spawn(function()
for _,b in ipairs(children)do
local playerName=b.Text
local plr=Players:FindFirstChild(playerName)
if plr and (not plr.Team or plr.Team.Name~=teamName)then
while sendingMessages and plr and (not plr.Team or plr.Team.Name~=teamName)do
channel:SendAsync(":team "..playerName.." "..teamName)
task.wait(0.1)
end
b.BackgroundColor3=Color3.fromRGB(255,0,0)
if plr and plr.Character and plr.Character:FindFirstChild("Highlight")then
plr.Character.Highlight.FillColor=Color3.fromRGB(255,0,0)
end
end
end
end)
end)
local btnSabitle
btnSabitle=createBtn("Sabitle",Color3.fromRGB(150,0,200),function()
if not isLocked then
isLocked=true
btnSabitle.Text="İptal"
else
isLocked=false
btnSabitle.Text="Sabitle"
updateList()
end
end)
createBtn("Kapat",Color3.fromRGB(200,0,0),function()
sendingMessages=false
isLocked=false
if updateConnection then
updateConnection:Disconnect()
updateConnection=nil
end
if scanningTask then
task.cancel(scanningTask)
scanningTask=nil
end
clearHighlights()
for _,b in ipairs(scroll:GetChildren())do
if b:IsA("TextButton")then
b:Destroy()
end
end
if bubble then
bubble:Destroy()
bubble=nil
end
if gui then
gui:Destroy()
gui=nil
end
end)
local dragging=false
local dragInput
local dragStart
local startPos
title.InputBegan:Connect(function(input)
if input.UserInputType==Enum.UserInputType.MouseButton1 then
dragging=true
dragStart=input.Position
startPos=main.Position
input:CaptureMouse()
end
end)
title.InputEnded:Connect(function(input)
if input.UserInputType==Enum.UserInputType.MouseButton1 then
dragging=false
if dragInput then
dragInput:ReleaseMouse()
dragInput=nil
end
end
end)
title.InputChanged:Connect(function(input)
if input.UserInputType==Enum.UserInputType.MouseMovement then
dragInput=input
end
end)
UserInputService.InputChanged:Connect(function(input)
if input==dragInput and dragging then
local delta=input.Position-dragStart
main.Position=UDim2.new(startPos.X.Scale,math.clamp(startPos.X.Offset+delta.X,0,workspace.CurrentCamera.ViewportSize.X-main.AbsoluteSize.X),startPos.Y.Scale,math.clamp(startPos.Y.Offset+delta.Y,0,workspace.CurrentCamera.ViewportSize.Y-main.AbsoluteSize.Y))
end
end)
updateConnection=RunService.RenderStepped:Connect(function()
if bubble then
bubble.Position=HRP.Position
end
end)
scanningTask=task.spawn(function()
while task.wait(scanInterval)do
updateList()
end
end)
